#define max 30
#include<string.h>
int top=-1;
int stack[max];


char pop()
{
  if(top==-1)
  cout<< "stack underflow";
else
 return (stack[top--]);
}

int match(char a ,char b)
{
   if(a=='['  &&  b==']')
   return 1;
  if(a=='('  && b==')')
  return 1;
  if(a=='{' && b=='}')
   return 1;
}

void push(char item)
{
  if(top==max-1)
   cout<<"stack overflow";
else
  top++;
 stack[top]=item;
}
